\subsection{$k[X_1,\ldots,X_n]$ est noethérien}
\begin{frame}

Soit $A$ un anneau commutatif. \\
2 conditions équivalentes pour que $A$ soit un anneau noethérien  : 

\begin{itemize}
	\item tout idéal $I$ de $A$ est engendré par un nombre fini d'éléments.

\item Une suite croissante d'idéaux d'un anneau noethérien est stationnaire.
\end{itemize}


\end{frame}
\begin{frame}
\begin{Thm}[de la base de Hilbert]
Soit $A$ un anneau noethérien. Alors $A[X]$ est aussi un anneau noethérien.
\end{Thm}

Conséquence immédiate : $k[X_1,\ldots,X_n]$ est un anneau noethérien.


\end{frame}

Ordre monomial = relation de bon ordre total de $\mathscr{M}$ compatible avec le produit 

%\begin{enumerate}

Pour $P:=\sum_{\alpha} p_{\alpha}X^\alpha \in k[X_1,\ldots,X_n]$ et $\geq$ un ordre monomial 
\begin{enumerate}
\item  Monôme dominant de $P$ : $\LM(P):=\max\{X^\alpha \in \mathscr{M} | p_\alpha \neq 0\}$
\item Multidegré de $f$, noté $\multideg(P)$, défini par : $X^{\multideg(P)}=LM(P)$
 \item Coefficient dominant de $P$ : $\LC(P):=p_{\multideg(P)}$
\item Terme dominant de $P$ : $\LT(P):=\LC(P)\cdot \LM(P)$
\end{enumerate}

\end{frame}



\begin{frame}


Ordre lexicographique $\geq_{lex}$ : \\

$\alpha=(\alpha_1,\ldots,\alpha_n), \beta=(\beta_1,\ldots,\beta_n) \in \mathbb{N}^n$, \\
$X^\alpha \geq_{lex} X^\beta$ ssi $\alpha=\beta$ ou le premier coefficient non nul en lisant par la gauche de $\alpha -\beta $ est positif.\\

Dans $k[X,Y,Z]$, \\
$X^3>_{lex}X^2 >_{lex}X$\\
$X>_{lex}Y>_{lex}Z$\\
 $X^2YZ^2>_{lex}X^2Z^4 >_{lex}XY^2Z$\\

\end{frame}

\begin{frame}
\begin{algorithm}[H]
\scriptsize
\caption{Algorithme de division}

\begin{algorithmic}
\REQUIRE $f_1,\ldots,f_s,f$
\ENSURE $a_1,\ldots,a_s,r$
\STATE $a_1 :=0 ;\ldots ; a_s:=0 ; r:=0$
\STATE $p:=f$
\WHILE{$p \neq 0$}
\STATE $i:=1$
\STATE div:=false
\WHILE{$i \leq s $ et div = false }
\IF{$\LT (f_i) | \LT (p)$}
\STATE $a_i:= a_i+\LT (p)/\LT (f_i)$
\STATE $p:=p-(\LT(p)/\LT (f_i)) f_i$
\ELSE
\STATE $i:=i+1$
\ENDIF
\ENDWHILE
\IF{div=false}
\STATE $r:=r+\LT (p)$
\STATE $p:=p-\LT (p) $
\ENDIF
\ENDWHILE
\end{algorithmic}

\end{algorithm}
\end{frame}

\begin{frame}
\begin{algorithm}[H]

\caption{Algorithme de Buchberger}

\begin{algorithmic}

\REQUIRE $F=(f_1,\ldots,f_s)$

\ENSURE Une base de Gröbner $G=(g_1,\ldots,g_t)$ de $I $, avec $F \subset G $

\STATE $G:=F $

\REPEAT

\STATE $G':=G $

\FOR{chaque paire $\{p,q\}\in G'^2$, $p \neq q $}

\STATE $S:=\overline{S (p,q)}^{G'} $

\IF{$S \neq 0$}

\STATE $G:=G \cup \{S\} $

\ENDIF
\ENDFOR
\UNTIL $G=G'$

\end{algorithmic}

\end{algorithm}
\end{frame}

\section{Résolution de systèmes polynomiaux}

\subsection{Théorème d'élimination}
\begin{frame}

$I=\left\langle f_1,\ldots,f_s \right\rangle $ un idéal de $k[X_1,\ldots,X_n]$. \\
 $p$ème idéal d'élimination de $I$ :  $I_p:=k[X_{p+1},\ldots,X_n] \cap I$

\begin{Thm}[d'élimination]
Soit $I $ un idéal de $k[X_1,\ldots,X_n]$ et $G$ une base de Gröbner de $I$ selon l'ordre lexicographique. Alors, pour tout $p \in [\![0,n]\!]$, l'ensemble $G_p =G \cap k[X_{p+1},\ldots,X_n]  $ est une base de Gröbner du $p$ème idéal d'élimination $I_p$.
\end{Thm}


Avec les $f,g,h$ de l'exemple précédent, 
\\$I_1=\left\langle X - YZ^2 + Y, Y^2 + 2*Z^2 - 5, Z^4 - 7/2Z^2 + 3\right\rangle$ \\ 
$I_2=\left\langle Z^4 - 7/2Z^2 + 3\right\rangle$


\end{frame}

\subsection{Résultant}
\begin{frame}
Soient $f=\sum_{k=1}^p f_k X_i^k,g=\sum_{k=1}^q g_k X_i^k \in k[X_1,\ldots,X_n]$ avec $a_k,b_k \in k[X_1,\ldots,X_{i-1},X_{i+1},\ldots,X_n]$ et $a_p \neq 0 \neq b_q$. 
Le résultant en $X_i$ de $f$ et $g$  est : 
\scriptsize
$\Res_{X_i}(f,g)=\det\begin{pmatrix}
a_0 & 0 &\cdots & 0 & b_0 & 0 &0 & \cdots & 0 & 0 \\
a_1 & a_0& \ddots & 0 &b_1 & b_0 &0 & \ddots & 0 & 0 \\
\vdots & a_1 &\ddots & 0 &\vdots & b_1 &b_0 & \ddots & 0 & 0 \\
\vdots & \vdots &\ddots & a_0 &\vdots & \vdots & b_1 & \ddots & 0 & 0 \\
a_ {p-1} & \vdots& \ddots & a_1 &b_q & \vdots &\vdots & \ddots & b_0 & 0 \\
a_p & a_{p-1} &\ddots & \vdots &0 & b_q &\vdots & \ddots & b_1 & b_0 \\
0 & a_p &\ddots & \vdots &0 & 0 &b_q & \ddots & \vdots & b_1 \\
\vdots & \vdots &\ddots & \vdots &\vdots & \vdots &0 & \ddots & \vdots & \vdots \\
\vdots & \vdots &\ddots & a_{p-1} &\vdots & \vdots &0 & \ddots & b_q & \vdots \\
0 & 0& \ldots & a_p &0 & 0 &0 & \cdots & 0 & b_q \\
\end{pmatrix}$. \\
\end{frame}

\subsection{Théorème d'extension}
\begin{frame}
\begin{Thm}[d'extension]

 Soit $I=\left \langle f_1,\ldots,f_s \right\rangle$ un idéal de $\C[X_1,\ldots,X_n]$ et $I_1$ le premier idéal d'élimination.\\
Ecrivons, pour $i \in [\![1,s]\!]$, $f_i$ sous la forme :
$f_i=g(X_2,\ldots,X_n)X_1^{N_i}$+termes de degré $<N_i$ en $X_1$\\
où $N_i \geq 0$ et $g_i \in \mathbb{C}[X_2,\ldots,X_n]$ non nul si $f_i \neq 0$ ($g_i=0$ si $f_i=0$). \\
Supposons qu'on ait une solution partielle $(a_2,\ldots,a_n) \in Z(I_1)$. Si $(a_2,\ldots,a_n) \notin Z(g_1,\ldots,g_s)$ alors il existe $a_1 \in \mathbb{C}$ tel que $(a_1,\ldots,a_n) \in Z(I)$.
\end{Thm}
\end{frame}

\subsubsection{Géométrie de l'élimination}
\begin{frame}


$V=Z(f_1,\ldots,f_s) \subset \C^n $, $I_p$ les idéaux d'élimination de $\left\langle f_1,\ldots,f_s\right\rangle$

 $\pi_p (V) \subset Z (I_p) $.


\begin{Thm}  Soit $g_i $ défini dans le théorème d'extension. Alors : \\
$Z (I_1)=\pi(V) \cup (Z (g_1,\ldots,g_s) \cap Z (I_1))$
\end{Thm}


\end{frame}
\begin{frame}


\begin{Thm}[de fermeture]

 Soit $V=Z(f_1,\ldots,f_s) \subset \C^n$ et soit $I_p$ le $p$ème idéal d'élimination de $\left\langle f_1,\ldots,f_s \right\rangle$. Alors  
\begin{enumerate}
\item $Z(I_p)$ est la plus petite (au sens de l'inclusion) variété contenant $\pi_p(V)$

\item Si $V \neq \emptyset$, alors il existe une variété affine $W \subsetneqq Z(I_p)$ telle que $Z(I_p) \setminus W \subset \pi_p(V)$
\end{enumerate}
\end{Thm}

\end{frame}

\begin{frame}


\begin{Thm}
Soit $F=(f_1,\ldots,f_n):k^m \to k^n$ une application à composantes polynomiales .\\ Soit $I$ l'id\'eal $\left\langle X_1-f_1,\ldots,X_n-f_n \right\rangle \subset k[T_1,\ldots,T_m,X_1,\ldots,X_n]$ et $I_m$ son $m$ \`eme id\'eal d'\'elimination. Alors $Z(I_m)$ est la plus petite variété de $k^n$ contenant $F(k^m)$.  

\end{Thm}\end{frame}
