\subsection{Généralités} 
\begin{Def}

 Un graphe non orienté est un couple $(S,A)$, où $S$ est un ensemble fini non vide (des éléments sont les sommets ) et $A$ est une partie de l'ensemble $\mathcal{P}_2 (S) $ des paires d'éléments de $S $ (les éléments de $A $ sont les arêtes). 
\end{Def}  
\begin{Def} 
Soit $G:=(A,S)$ un graphe non orienté. Les sommets $s,t$ sont dits adjacents si $(s,t) \in A$ 
\end{Def}  
\begin{Def} 
Soit $p\in \mathbb{N}^*$.\\ Notons $C_p=\{x_1,\ldots,x_p\}$ un ensemble de couleurs. Un graphe $G:=(A,S)$ est coloriable si on peut associer à chaque sommet de $G$ une couleur de $C_p$ tel que deux sommets adjacents n'aient pas la même couleur.  
\end{Def}
  \subsection{Equations polynomiales}

	Soit $n:=Card(A)$.\\
