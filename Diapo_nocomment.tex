\section{Description des idéaux de $k[X_1,\ldots,X_n]$}

\subsection{Cas $k[X]$}

\begin{frame}

\begin{itemize}
	\item $k[X]$ est un anneau euclidien et donc principal. \\
\vspace{\baselineskip} 
\pause
\item Idéal de $k[X]$ de la forme $Pk[X]$. \\
\vspace{\baselineskip}
\pause 
\item Division euclidienne de $f$ par $P$ : $f=PQ+R$ où $\deg(R) < \deg(P)$.\\
\vspace{\baselineskip} 
\pause
\item Caractérisation d'appartenance à $I$ :  $f \in I \Leftrightarrow R=0$.
\end{itemize}

\end{frame}



\begin{frame}

$k[X_1,\ldots,X_n]$ n'est pas principal pour $n>1$.

\subsection{$k[X_1,\ldots,X_n]$ est noethérien}

\begin{frame}

Soit $A$ un anneau commutatif. \\
\vspace{2\baselineskip} 
2 conditions équivalentes pour que $A$ soit un anneau noethérien  : 
\vspace{\baselineskip} 
\pause
\begin{itemize}
	\item tout idéal $I$ de $A$ est engendré par un nombre fini d'éléments.
\vspace{\baselineskip} 
\pause
\item Une suite croissante d'idéaux d'un anneau noethérien est stationnaire.
\end{itemize}


\end{frame}
\begin{frame}
\begin{Thm}[de la base de Hilbert]
Soit $A$ un anneau noethérien. Alors $A[X]$ est aussi un anneau noethérien.
\end{Thm}
\pause
\vspace{2\baselineskip} 
\begin{itemize}
	\item Conséquence immédiate : $k[X_1,\ldots,X_n]$ est un anneau noethérien.
\end{itemize}



\end{frame}

\subsection{Ordres monomiaux}
\begin{frame}
\subsubsection{Définitions}

Ordre monomial = relation de bon ordre total sur les monômes compatible avec le produit 
\pause

\vspace{2\baselineskip}
Pour $P:=\sum_{\alpha} p_{\alpha}X^\alpha \in k[X_1,\ldots,X_n]$ et $\geq$ un ordre monomial 
 \vspace{\baselineskip} 
\begin{enumerate}
\item  Monôme dominant de $P$ : $\LM(P):=\max\{X^\alpha \in \mathscr{M} | p_\alpha \neq 0\}$
\item Multidegré de $f$, noté $\multideg(P)$, défini par : $X^{\multideg(P)}=\LM(P)$
 \item Coefficient dominant de $P$ : $\LC(P):=p_{\multideg(P)}$
\item Terme dominant de $P$ : $\LT(P):=\LC(P)\cdot \LM(P)$
\end{enumerate}

\end{frame}



\begin{frame}


Ordre lexicographique $\geq_{lex}$ : \\
\vspace{\baselineskip} 
$X^\alpha \geq_{lex} X^\beta$ ssi $\alpha=\beta$ ou le premier coefficient non nul en lisant par la gauche de $\alpha -\beta $ est positif.\\
\vspace{2\baselineskip} 
\pause
Dans $k[X,Y,Z]$, \\
$X^3>_{lex}X^2 >_{lex}X$\\
$X>_{lex}Y>_{lex}Z$\\
 $X^2YZ^2>_{lex}X^2Z^4 >_{lex}XY^2Z$\\

\end{frame}


\subsection{Algorithme de division}


\begin{frame}
\begin{algorithm}[H]
\scriptsize
\caption{Algorithme de division}

\begin{algorithmic}
\REQUIRE $f_1,\ldots,f_s,f$
\ENSURE $a_1,\ldots,a_s,r$
\STATE $a_1 :=0 ;\ldots ; a_s:=0 ; r:=0$
\STATE $p:=f$
\WHILE{$p \neq 0$}
\STATE $i:=1$
\STATE div:=false
\WHILE{$i \leq s $ et div = false }
\IF{$\LT (f_i) | \LT (p)$}
\STATE $a_i:= a_i+\LT (p)/\LT (f_i)$
\STATE $p:=p-(\LT(p)/\LT (f_i)) f_i$
\ELSE
\STATE $i:=i+1$
\ENDIF
\ENDWHILE
\IF{div=false}
\STATE $r:=r+\LT (p)$
\STATE $p:=p-\LT (p) $
\ENDIF
\ENDWHILE
\end{algorithmic}

\end{algorithm}
\end{frame}

\section{Bases de Gröbner}
\begin{frame}

\subsection{Généralités}


$I$ un idéal de $k[X_1,\ldots,X_n]$  non réduit à $\{0\}$ \\

\vspace{3\baselineskip} 
\pause
Base de Gröbner : $G=\{g_1,\ldots,g_s\} \subset I$ tel que : \\
$\left\langle \LT(I) \right\rangle=\left\langle \LT(g_1),\ldots,\LT(g_s)\right\rangle$.



\end{frame}
\begin{frame}

\begin{itemize}
	\item Tout idéal a une base de Gröbner 
	\vspace{3\baselineskip} 
	\pause
\item Une base de Gröbner est une base de l'idéal  \\
\vspace{3\baselineskip} 
\pause
\item $f \in I$ ssi le reste est nul
\end{itemize}

 \end{frame}
 

\begin{frame}
 Soient $f,g \in k[X_1,\ldots,X_n]$ des polynômes non nuls.

Le $S$-polynôme de $f$ et $g$ est le polynôme : $S(f,g):=\frac{X^\gamma}{\LT(f)}f-\frac{X^\gamma}{\LT(g)} g$\\ où les coordonnées de $\gamma$ sont les maxima des coordonnées de $\multideg(f)$ et $\multideg(g)$

\pause
\begin{Thm} Soit $I$ un idéal de $k[X_1,\ldots,X_n]$. Alors une base $G=\{g_1,\ldots,g_s\}$ de $I$ est une base de Gröbner de $I$ si, et seulement si, pour tout couple $(i,j)$, $i \neq j$, $\overline{S(g_i,g_j)}^G=0$
\end{Thm}

\end{frame}

\subsection{Algorithme de Buchberger}


\begin{frame}
\begin{algorithm}[H]

\caption{Algorithme de Buchberger}

\begin{algorithmic}

\REQUIRE $F=(f_1,\ldots,f_s)$

\ENSURE Une base de Gröbner $G=(g_1,\ldots,g_t)$ de $I $, avec $F \subset G $

\STATE $G:=F $

\REPEAT

\STATE $G':=G $

\FOR{chaque paire $\{p,q\}\in G'^2$, $p \neq q $}

\STATE $S:=\overline{S (p,q)}^{G'} $

\IF{$S \neq 0$}

\STATE $G:=G \cup \{S\} $

\ENDIF
\ENDFOR
\UNTIL $G=G'$

\end{algorithmic}

\end{algorithm}
\end{frame}


\section{Résolution de systèmes polynomiaux}

\subsection{Théorème d'élimination}

\begin{frame}
Avec les $f,g,h$ de l'exemple précédent,  
$I=\left\langle X - YZ^2 + Y, Y^2 + 2Z^2 - 5, Z^4 - 7/2Z^2 + 3\right\rangle$ \\
\pause
\vspace{1\baselineskip} 
$I=\left\langle f_1,\ldots,f_s \right\rangle $ un idéal de $k[X_1,\ldots,X_n]$. \\
 $p$ème idéal d'élimination de $I$ :  $I_p:=k[X_{p+1},\ldots,X_n] \cap I$
\pause
\begin{Thm}[d'élimination]
Soit $I $ un idéal de $k[X_1,\ldots,X_n]$ et $G$ une base de Gröbner de $I$ selon l'ordre lexicographique. Alors, pour tout $p \in [\![0,n]\!]$, l'ensemble $G_p =G \cap k[X_{p+1},\ldots,X_n]  $ est une base de Gröbner du $p$ème idéal d'élimination $I_p$.
\end{Thm}
\pause

$I_1=\left\langle  Y^2 + 2Z^2 - 5, Z^4 - 7/2Z^2 + 3\right\rangle$ \\ 
$I_2=\left\langle Z^4 - 7/2Z^2 + 3\right\rangle$

\end{frame}


\subsection{Géométrie}
\begin{frame}


\begin{itemize}
	\item Variété affine définie par $f_1,\ldots,f_s$ : \\
$Z(f_1,\ldots,f_s)=\{a \in k^n | \forall i \in [\![1,s]\!],f_i(a)=0 \}$.
\vspace{3\baselineskip} 
\pause
\item Intersection de deux variétés est une variété. \\
\vspace{3\baselineskip} \pause
\item Si $\left\langle \mathcal{F} \right\rangle=\left\langle \mathcal{G} \right\rangle$ alors $Z(\mathcal{F})=Z(\mathcal{G})$ \\
\vspace{\baselineskip} \pause
En particulier, $Z(f_1,\ldots,f_s)=Z(\left\langle f_1,\ldots,f_s\right\rangle)$
\end{itemize}
\end{frame}


\subsection{Théorème d'extension}
\begin{frame}
\begin{Thm}[d'extension]

 Soit $I=\left \langle f_1,\ldots,f_s \right\rangle$ un idéal de $\C[X_1,\ldots,X_n]$ et $I_1$ le premier idéal d'élimination.\\
Ecrivons, pour $i \in [\![1,s]\!]$, $f_i$ sous la forme :
$f_i=g(X_2,\ldots,X_n)X_1^{N_i}$+termes de degré $<N_i$ en $X_1$\\
où $N_i \geq 0$ et $g_i \in \mathbb{C}[X_2,\ldots,X_n]$ non nul si $f_i \neq 0$ ($g_i=0$ si $f_i=0$). \\
Supposons qu'on ait une solution partielle $(a_2,\ldots,a_n) \in Z(I_1)$. Si $(a_2,\ldots,a_n) \notin Z(g_1,\ldots,g_s)$ alors il existe $a_1 \in \mathbb{C}$ tel que $(a_1,\ldots,a_n) \in Z(I)$.
\end{Thm}
\end{frame}


\begin{frame}

Reprenons l'exemple précédent : \\ 
\vspace{2\baselineskip} \pause
$I_1=\left\langle Y^2 + 2Z^2 - 5,Z^4-7/2Z^2+3 \right\rangle$ et $I_2=\left\langle Z^4-7/2Z^2+3 \right\rangle$. \\
\vspace{\baselineskip} \pause
$Z(I_2)=Z(\left\langle Z^4-7/2Z^2+3\right\rangle)=\{\pm \sqrt{2},\pm \sqrt{3/2}\}$. \\
\vspace{\baselineskip} \pause
$Z(I_1)=\{(\pm 1,\pm \sqrt{2}),(\pm\sqrt{2},\pm \sqrt{3/2})\}$ \\
\vspace{\baselineskip}\pause 
$Z(I)=\{\pm(1,1,\pm\sqrt{2}),\pm(1/\sqrt{2},\sqrt{2},\pm\sqrt{2})\}$

\end{frame}
\subsection{Géométrie de l'élimination}




\begin{frame}


$V=Z(f_1,\ldots,f_s) \subset \C^n $, $I_p$ les idéaux d'élimination de $\left\langle f_1,\ldots,f_s\right\rangle$ \\

\vspace{3\baselineskip} 
\begin{Thm}[d'extension version géométrique]  Soit $g_i $ défini dans le théorème d'extension. Alors : \\
$Z (I_1)=\pi(V) \cup (Z (g_1,\ldots,g_s) \cap Z (I_1))$
\end{Thm}


\end{frame}
\begin{frame}


\begin{Thm}[de fermeture]

 Soit $V=Z(f_1,\ldots,f_s) \subset \C^n$ et soit $I_p$ le $p$ème idéal d'élimination de $\left\langle f_1,\ldots,f_s \right\rangle$. Alors  

$Z(I_p)$ est la plus petite (au sens de l'inclusion) variété contenant $\pi_p(V)$

\end{Thm}

\end{frame}
\section{Implicitation}

\begin{frame}


\begin{Thm}
Soit $F=(f_1,\ldots,f_n):k^m \to k^n$ une application à composantes polynomiales .\\ Soit $I$ l'id\'eal $\left\langle X_1-f_1,\ldots,X_n-f_n \right\rangle \subset k[T_1,\ldots,T_m,X_1,\ldots,X_n]$ et $I_m$ son $m$ \`eme id\'eal d'\'elimination. Alors $Z(I_m)$ est la plus petite variété de $k^n$ contenant $F(k^m)$.  

\end{Thm}\end{frame}
